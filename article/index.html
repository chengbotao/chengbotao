    URI针对特殊字符编码处理

    encodeURIC

    URI 针对特殊字符编码处理

    实现

    export function encodeURI(val: string): string {
	return encodeURIComponent(val)
		.replace(/%40/g, "@")
		.replace(/%3A/gi, ":")
		.replace(/%24/g, "$")
		.replace(/%2C/gi, ",")
		.replace(/%20/g, "+")
		.replace(/%5B/gi, "[")
		.replace(/%5D/gi, "]");
}
